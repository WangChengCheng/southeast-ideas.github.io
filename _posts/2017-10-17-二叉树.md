### 二叉树
二叉树是一种树型结构，它的特点是每个结点至多只有两棵子树，而且二叉树的子树有左右之分，其次序不能任意颠倒。
### Code

```java
public class BinaryTree<T extends Comparable<T>> {
    public T data;
    public BinaryTree left;
    public BinaryTree right;

    public BinaryTree() {
    }

    public BinaryTree(T data) {
        this.data = data;
    }

    public static void main(String[] args){
        BinaryTree<Integer> binTree = new BinaryTree<>();
        binTree.insert(10);
        binTree.insert(23);
        binTree.insert(5);
        binTree.insert(6);
        binTree.insert(19);
        binTree.insert(4);
        binTree.insert(30);
        System.out.print("DLR: ");
        dlr(binTree);
        System.out.println();
        System.out.print("LDR: ");
        ldr(binTree);
        System.out.println();
        System.out.print("LRD: ");
        lrd(binTree);
        System.out.println();
    }

    public boolean insert(T i) {
        if (data == null) {
            data = i;
            return true;
        }
        BinaryTree current = this;
        while (true) {
            if (i.compareTo((T)current.data) < 0) {//if i is less than current node
                if (current.left != null) {
                    current = current.left;
                } else {
                    current.left = new BinaryTree(i);
                    break;
                }
            } else {//if i is bigger than current node
                if (current.right != null) {
                    current = current.right;
                } else {
                    current.right = new BinaryTree(i);
                    break;
                }
            }
        }
        return true;
    }

    /**
     * 前序遍历(DLR)
     */
    public static void dlr(BinaryTree binTree) {
        if (binTree != null) {
            System.out.print(binTree.data+", ");
            if (binTree.left != null) {
                dlr(binTree.left);
            }
            if (binTree.right != null) {
                dlr(binTree.right);
            }
        }else
            System.out.println("空树!");
    }

    /**
     * 中序遍历(LDR)
     */
    public static void ldr(BinaryTree binTree) {
        if (binTree != null) {
            if (binTree.left != null) {
                ldr(binTree.left);
            }
            System.out.print(binTree.data+", ");
            if (binTree.right != null) {
                ldr(binTree.right);
            }
        }else
            System.out.println("空树!");
    }

    /**
     * 后序遍历（LRD）
     */
    public static void lrd(BinaryTree binTree) {
        if (binTree != null) {
            if (binTree.left != null) {
                lrd(binTree.left);
            }
            if (binTree.right != null) {
                lrd(binTree.right);
            }
            System.out.print(binTree.data+", ");
        }else
            System.out.println("空树!");
    }
}
```
